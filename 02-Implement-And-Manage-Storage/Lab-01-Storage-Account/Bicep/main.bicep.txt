// === PARAMETERS ===
// These are the inputs for our template.
@description('The name of the Azure Storage account. Must be globally unique.')
param storageAccountName string = 'tfmbicep11082025'

@description('The location where the resources will be deployed.')
param location string = 'EastUS'


// === RESOURCES ===
// This is the actual resource we are telling Azure to build.
resource storageAccount 'Microsoft.Storage/storageAccounts@2021-09-01' = {
  name: storageAccountName
  location: location
  sku: {
    name: 'Standard_LRS'
  }
  kind: 'StorageV2'
}


// === OUTPUTS ===
// This will display the name of the storage account after deployment.
output storageAccountName string = storageAccount.name
